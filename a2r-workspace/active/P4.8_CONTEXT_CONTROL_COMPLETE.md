## Core API

### ContextControlPlane

```rust
use a2r_context_control::{ContextControlPlane, ContextResolution};
use a2r_memory_kernel::MemoryKernel;

// Create control plane
let memory = Arc::new(MemoryKernel::default());
let cp = ContextControlPlane::new(memory);

// Create context
let context_id = cp.create_context("Project Alpha").await;

// Commit to context
let commit_id = cp
    .commit(&context_id, "main", "Initial commit", "content")
    .await?;

// Create branch
let branch_id = cp
    .create_branch(&context_id, "main", "feature")
    .await?;

// Merge branch
cp.merge(&context_id, "feature", "main", "fast-forward").await?;

// Get context at resolution
let snapshot = cp
    .get_context_at(&context_id, ContextResolution::Summary)
    .await?;

// Export bundle
let bundle = cp.export_bundle(&context_id, "main").await?;
```
