## Code Changes Required

### 1. Remove API Key Dependency

**Before:**
```typescript
// Original agentation
import { sendToAPI } from './api';

export function useAgentation() {
  const apiKey = process.env.AGENTATION_API_KEY;
  
  async function saveAnnotation(annotation) {
    await sendToAPI(apiKey, annotation);
  }
}
```

**After:**
```typescript
// A2R absorbed version
export function useAgentation() {
  async function saveAnnotation(annotation) {
    // Store in localStorage (dev-only)
    const existing = JSON.parse(localStorage.getItem('agentation') || '[]');
    existing.push(annotation);
    localStorage.setItem('agentation', JSON.stringify(existing));
  }
}
```

### 2. Add NODE_ENV Gate

```typescript
// 6-ui/a2r-platform/src/dev/agentation/index.ts

if (process.env.NODE_ENV === 'production') {
  // Never load in production
  console.warn('Agentation is dev-only. Skipping initialization.');
  export const AgentationOverlay = () => null;
} else {
  // Dev-only initialization
  export { AgentationOverlay } from './components/AgentationOverlay';
}
```

### 3. Add A2R Export Adapter

```typescript
// 6-ui/a2r-platform/src/dev/agentation-adapter.ts

export interface AgentationOutput {
  notes: string;
  selectors: string[];
  context: string;
  screenshot?: string;
}

export function copyToClipboard(output: AgentationOutput) {
  // Original Agentation copies JSON
  const json = JSON.stringify(output, null, 2);
  navigator.clipboard.writeText(json);
}

export function copyForA2R(output: AgentationOutput, header: A2RExecutionHeader) {
  // A2R adapter copies formatted markdown
  const formatted = formatAgentInstructions(output, header);
  navigator.clipboard.writeText(formatted);
}
```
