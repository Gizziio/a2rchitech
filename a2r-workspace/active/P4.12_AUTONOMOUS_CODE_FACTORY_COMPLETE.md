## Core API

### AutonomousCodeFactory

```rust
use a2r_autonomous_code_factory::{
    AutonomousCodeFactory, CodeGenerationRequest, ValidationResult
};

// Create factory
let factory = AutonomousCodeFactory::new(
    context_plane, memory, system_law, harness, evolution
);

// Generate code from spec
let request = CodeGenerationRequest {
    request_id: "req_001".to_string(),
    spec_ref: "spec/api".to_string(),
    context_id: ctx_id.clone(),
    requirements: vec!["REST API".to_string()],
    constraints: vec!["No blocking".to_string()],
    acceptance_tests: vec!["test_api()".to_string()],
};

let code = factory.generate_code(request).await?;

// Validate generated code
let validation = factory
    .validate_code(&code.artifact_id, &code.code, &code.tests)
    .await?;

// Make merge decision
let decision = factory
    .make_merge_decision(&code.artifact_id, &validation, &ctx_id)
    .await?;

assert!(decision.approved);
```
