## Core API

```rust
use a2r_quality_score::{QualityScoreEngine, DomainGrade, EntropyScore};

// Create engine
let engine = QualityScoreEngine::new(system_law);

// Update domain grade
let grade = DomainGrade {
    domain: "api".to_string(),
    architecture_adherence: 0.9,
    test_coverage: 0.85,
    observability_completeness: 0.8,
    boundary_enforcement: 0.95,
    drift_frequency: 0.1,
    overall_grade: 'A',
    last_updated: Utc::now(),
};
engine.update_domain_grade(grade).await;

// Record entropy
let mut entropy = EntropyScore {
    score_id: "test".to_string(),
    rule_violations: 5,
    drift_rate: 0.2,
    test_coverage_delta: -0.05,
    documentation_mismatch: 0.1,
    total_score: 0.0,
    timestamp: Utc::now(),
};
engine.record_entropy(entropy).await;

// Get status
let status = engine.get_status().await;
println!("Trend: {:?}", status.trend);
println!("Recommendations: {:?}", status.recommendations);
```
