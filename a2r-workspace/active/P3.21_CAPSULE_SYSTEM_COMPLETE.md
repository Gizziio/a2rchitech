## Core API

### CapsuleRegistry

```rust
use a2r_capsule_sdk::{CapsuleRegistry, RunReference};
use a2rchitech_system_law::SystemLawEngine;

// Create registry with SYSTEM_LAW integration
let system_law = Arc::new(SystemLawEngine::new());
let registry = CapsuleRegistry::new(system_law);

// Register framework
let framework = CapsuleFramework {
    framework_id: "fwk_diff_review".to_string(),
    version: "1.0.0".to_string(),
    // ... configure framework
};
registry.register_framework(framework).await;

// Spawn capsule from framework
let run_ref = RunReference {
    run_id: "run_001".to_string(),
    session_id: "sess_001".to_string(),
};

let capsule = registry
    .spawn_capsule("fwk_diff_review", "review this diff", run_ref)
    .await?;

// Get capsule
let capsule = registry.get_capsule(&capsule.capsule_id).await;

// Update capsule
let update = CapsuleUpdate {
    bindings: Some(new_bindings),
    ..Default::default()
};
registry.update_capsule(&capsule.capsule_id, update).await?;

// Pin capsule
registry.pin_capsule(&capsule.capsule_id).await?;

// Close capsule
registry.close_capsule(&capsule.capsule_id, CloseMode::ArchiveToJournal).await?;

// List active capsules
let active = registry.list_active_capsules().await;
```
