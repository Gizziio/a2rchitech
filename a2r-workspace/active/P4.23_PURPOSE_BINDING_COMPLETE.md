## Hard Gates

### Gate 1: Purpose Required
```rust
if purpose.trim().is_empty() {
    return Err(PurposeBindingError::PurposeRequired);
}
```

### Gate 2: Tool Allowlist
```rust
if !scope.allowed_tools.is_empty() && !scope.allowed_tools.contains(&tool_id) {
    return Err(PurposeBindingError::ToolNotAllowed(tool_id));
}
```

### Gate 3: Tool Denylist
```rust
if scope.denied_tools.contains(&tool_id) {
    return Err(PurposeBindingError::ToolDenied(tool_id));
}
```

### Gate 4: Budget Constraints
```rust
if let Some(max_budget) = scope.max_budget {
    if max_budget <= 0.0 {
        return Err(PurposeBindingError::BudgetExceeded);
    }
}
```

## Usage Example

```rust
// Initialize engine
let engine = PurposeBindingEngine::new(system_law, harness);

// Before tool execution
async fn execute_tool_with_purpose(
    engine: &PurposeBindingEngine,
    wih_id: &str,
    tool_id: &str,
    purpose: &str,
) -> Result<(), PurposeBindingError> {
    // Bind purpose
    let binding = engine.bind_purpose(wih_id, tool_id, purpose, "agent_001").await?;
    
    // Validate
    let result = engine.validate_purpose(&binding.binding_id).await?;
    if !result.valid {
        return Err(PurposeBindingError::PurposeOutOfScope(result.errors.join(", ")));
    }
    
    // Mark as validated
    engine.mark_validated(&binding.binding_id).await?;
    
    // Execute tool
    execute_tool(tool_id).await?;
    
    // Mark as completed
    engine.mark_completed(&binding.binding_id).await?;
    
    Ok(())
}
```
