# Integrate Automerge CRDT for Sync

## Objective
Add conflict-free document sync using Automerge.

## Implementation

### CRDT Document
```typescript
import * as Automerge from "@automerge/automerge";

interface WorkspaceState {
  tasks: Task[];
  sessions: Session[];
  skills: Skill[];
  settings: Settings;
}

export class WorkspaceDocument {
  private doc: Automerge.Doc<WorkspaceState>;
  
  constructor() {
    this.doc = Automerge.init<WorkspaceState>();
  }
  
  change(fn: (state: WorkspaceState) => void) {
    this.doc = Automerge.change(this.doc, fn);
  }
  
  merge(other: Automerge.Doc<WorkspaceState>) {
    this.doc = Automerge.merge(this.doc, other);
  }
  
  save(): Uint8Array {
    return Automerge.save(this.doc);
  }
}
```
