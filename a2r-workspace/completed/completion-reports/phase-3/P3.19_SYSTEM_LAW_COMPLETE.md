## Core API

### SystemLawEngine

```rust
// Create new engine
let engine = SystemLawEngine::new();

// Record violation
let violation_id = engine.record_violation(
    "LAW-GRD-001",
    ViolationSeverity::Hard,
    "Unvalidated assumption",
    serde_json::json!({"assumption": "test"}),
).await;

// Check for hard violations (must halt execution)
if engine.has_hard_violations().await {
    // Halt execution
}

// Resolve violation
engine.resolve_violation(&violation_id, "Fixed").await;

// Get compliance report
let report = engine.get_compliance_report().await;
```

## Usage Examples

### Validate Entity Against Ontology

```rust
let engine = SystemLawEngine::new();

// Register entity type
let entity_type = EntityType {
    type_id: "agent".to_string(),
    name: "Agent".to_string(),
    description: "Autonomous agent".to_string(),
    required_fields: vec!["id".to_string(), "role".to_string()],
    optional_fields: vec!["name".to_string()],
    relationships: vec![],
    version: "1.0.0".to_string(),
};
engine.register_entity_type(entity_type).await;

// Validate entity
let mut fields = HashMap::new();
fields.insert("id".to_string(), serde_json::json!("agent_001"));
fields.insert("role".to_string(), serde_json::json!("implementer"));

match engine.validate_entity("agent", &fields).await {
    Ok(_) => println!("Entity valid"),
    Err(violation) => println!("Entity invalid: {}", violation.description),
}
```

### Track Execution Session
