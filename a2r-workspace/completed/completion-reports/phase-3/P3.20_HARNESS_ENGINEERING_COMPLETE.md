## Core API

### HarnessEngineeringEngine

```rust
use a2rchitech_system_law::SystemLawEngine;
use a2rchitech_harness_engineering::HarnessEngineeringEngine;

// Create engine with SYSTEM_LAW integration
let system_law = Arc::new(SystemLawEngine::new());
let engine = HarnessEngineeringEngine::new(system_law);

// Set risk policy
engine.set_risk_policy(risk_policy).await;

// Run preflight
let preflight = engine.run_preflight("wih_001", changed_files).await;
if !preflight.allowed {
    // Halt execution
}

// Store evidence
engine.store_evidence(evidence_manifest).await;

// Validate evidence
let validation = engine.validate_evidence("ev_001", 24).await;

// Generate merge receipt
let receipt = engine.generate_merge_receipt(
    "abc123",  // head_sha
    RiskTier::Medium,
    "ev_hash",
    "review_001",
    "acceptance_hash",
).await;

// Run merge gates
let gate_result = engine.run_merge_gates(&receipt).await;
if gate_result.auto_merge_allowed {
    // Auto-merge
} else if gate_result.human_approval_required {
    // Require human approval
}

// Run garbage collection
let gc_result = engine.run_garbage_collection().await;
for action in gc_result.actions {
    // Process GC action
}
```

## Usage Examples

### Preflight Check

```rust
let changed_files = vec![
    ChangedFile {
        path: "src/main.rs".to_string(),
        status: FileStatus::Modified,
        additions: 50,
        deletions: 10,
    },
    ChangedFile {
        path: "spec/api.md".to_string(),
        status: FileStatus::Modified,
        additions: 20,
        deletions: 5,
    },
];

let preflight = engine.run_preflight("wih_001", changed_files).await;

assert!(preflight.allowed);
assert_eq!(preflight.risk_tier, RiskTier::Medium);
assert!(preflight.required_gates.contains(&"lint".to_string()));
assert!(preflight.required_gates.contains(&"tests".to_string()));
```

### Evidence Validation

```rust
let manifest = EvidenceManifest {
    manifest_id: "ev_001".to_string(),
    sha_embedded: true,
    flow_assertions: vec![
        FlowAssertion {
            assertion_id: "assert_001".to_string(),
            flow_name: "test".to_string(),
            passed: true,
            details: "All tests passed".to_string(),
        },
    ],
    artifacts: vec![
        EvidenceArtifact {
            artifact_id: "art_001".to_string(),
            artifact_type: "test_result".to_string(),
            path: "/test/results.xml".to_string(),
            sha_hash: "abc123".to_string(),
            timestamp: Utc::now(),
        },
    ],
    created_at: Utc::now(),
    verified: true,
};

engine.store_evidence(manifest).await;

let result = engine.validate_evidence("ev_001", 24).await;
assert!(result.valid);
assert!(result.sha_valid);
assert_eq!(result.assertions_passed, 1);
```

### Merge Gate Flow

```rust
// Generate receipt
let receipt = engine.generate_merge_receipt(
    "abc123",
    RiskTier::Medium,
    "ev_hash",
    "review_001",
    "acceptance_hash",
).await;

// Run gates
let gate_result = engine.run_merge_gates(&receipt).await;

if gate_result.allowed {
    if gate_result.auto_merge_allowed {
        println!("Auto-merge allowed");
    } else if gate_result.human_approval_required {
        println!("Human approval required");
    }
} else {
    for gate in &gate_result.gate_results {
        if !gate.passed {
            println!("Gate failed: {} - {}", gate.name, gate.message);
        }
    }
}
```
