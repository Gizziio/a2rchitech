## Core API

### VendorRegistry

```rust
use a2r_vendor_wrappers::{
    VendorRegistry, VendorWrapper, create_default_registry,
    FlexLayoutWrapper, CommandPaletteWrapper,
};

// Create default registry with all wrappers
let registry = create_default_registry();

// Get wrapper by ID
let wrapper = registry.get_wrapper("ui.flexlayout");
assert_eq!(wrapper.unwrap().wrapper_name, "FlexLayoutHost");

// Get wrappers by category
let ui_wrappers = registry.get_wrappers_by_category("ui");
assert_eq!(ui_wrappers.len(), 4); // flexlayout, panels, virtual, radix

let cmd_wrappers = registry.get_wrappers_by_category("cmd");
assert_eq!(cmd_wrappers.len(), 2); // palette, hotkeys

// Register custom wrapper
let mut registry = VendorRegistry::new();
registry.register(VendorWrapper {
    wrapper_id: "custom.wrapper".to_string(),
    vendor_package: "vendor-pkg".to_string(),
    vendor_version: "1.0.0".to_string(),
    wrapper_name: "CustomWrapper".to_string(),
    description: "Custom wrapper".to_string(),
    upstream_repo: Some("https://github.com/vendor/pkg".to_string()),
    license: "MIT".to_string(),
    exports: vec!["CustomWrapper".to_string()],
    usage_example: "<CustomWrapper />".to_string(),
    notes: None,
});
```
